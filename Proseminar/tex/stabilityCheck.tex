\begin{algorithm}[H]
  \SetKwData{KwM}{m}\SetKwData{KwW}{w}
  \FuerAlle{$\KwM \in M$}{
    \FuerJedes{$\KwW \in W . p_{P}(\KwM) <_{\KwM} \KwW$}{
      \Wenn{$p_{P}(\KwW)<_{\KwW} \KwM$}{
        \Return{instabile Paarung}\;
      }
    }
  }
  \Return{stabile Paarung}\;
  \caption{Stabilitätsprüfung}
\end{algorithm}
